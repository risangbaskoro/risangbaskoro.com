32 
33        return collect($this->client->getEntries($query)->getItems())

36                    'title' => $item->title,
37                    'content' => $item->content,

44    protected function imageFileToArray(?Asset $asset): ?array
45    {

50        /** @var ImageFile $imageFile */

55            'description' => $asset->getDescription(),

57            'url' => $imageFile->getUrl(),
